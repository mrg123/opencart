SELECT COUNT(DISTINCT p.product_id) AS total FROM oc_product p LEFT JOIN oc_product_description pd ON (p.product_id = pd.product_id) LEFT JOIN oc_product_to_store p2s ON (p.product_id = p2s.product_id) LEFT JOIN oc_ka_product_variants kpv ON (p.product_id = kpv.product_id) WHERE pd.language_id = '1' AND p.status = '1' AND p.date_available <= NOW() AND p2s.store_id = '0' AND ( pd.name LIKE '%product%' OR  kpv.model LIKE '%product%' OR  pd.tag LIKE '%product%' OR LCASE(p.model) = 'product' OR LCASE(p.sku) = 'product' OR LCASE(p.upc) = 'product' OR LCASE(p.ean) = 'product' OR LCASE(p.jan) = 'product' OR LCASE(p.isbn) = 'product' OR LCASE(p.mpn) = 'product')
